CREATE TABLE TB_MENSAGEM_PROTOCOLO_ISO8583
(
        ID_MENSAGEM CHAR(37),
        DT_CAPTURA CHAR(34),
        DT_ARMAZENAMENTO CHAR(34),
        IP_ORIGEM CHAR(15),
        PORTA_ORIGEM CHAR(10),
        ID_DESTINO CHAR(15),
        PORTA_DESTINO CHAR(10),        
        DS_MENSAGEM_PROTOCOLO VARCHAR(1000)
)
SEGMENTED BY HASH (ID_MENSAGEM)
ALL NODES

CREATE TABLE TB_ATRIBUTO_PROTOCOLO_ISO8583
(
        ID_MENSAGEM CHAR(37),
        ID_ATRIBUTO INT,
        DS_CONTEUDO VARCHAR(1000)
)
SEGMENTED BY HASH (ID_MENSAGEM)
ALL NODES
